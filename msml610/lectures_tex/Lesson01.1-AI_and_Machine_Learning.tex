  \item

  \item

  \item

  \item

    \item

    \end{itemize}
  \end{itemize}
\end{itemize}

\phantomsection\label{ai-formal-definition}
\begin{itemize}
\item
  AI is defined around \textbf{\textcolor{red}{two axes}}:

  \begin{itemize}
  \tightlist
  \item
    Thinking vs.~Acting
  \item
    Human vs.~Rational (ideal performance)
  \end{itemize}
\item
  Four possible definitions of AI as a machine that can:

  \begin{enumerate}
  \tightlist
  \item
    Think humanly
  \item
    Think rationally
  \item
    Act humanly
  \item
    Act rationally
  \end{enumerate}
\item
  \textbf{\textcolor{darkgreen}{Q}}: Which one do you think is the best

  We will see that building machines that can
  \textbf{\textcolor{blue}{"act rationally"}} should be ultimate goal of
  AI
\end{itemize}
\end{frame}

\begin{frame}{Thinking Rationally: Cons}
\phantomsection\label{thinking-rationally-cons}
\begin{enumerate}
\tightlist
\item
  \textbf{\textcolor{red}{Formalizing informal knowledge is difficult}}

  \begin{itemize}
  \tightlist
  \item
    Example: \emph{``A handshake occurs when two people extend, grip,
    shake hands, then release.''}
  \item
    Formal logic representation: \begin{align*}
        &\exists x, y \, ( \text{Person}(x) \land \text{Person}(y) \land x \neq y \land \\
        &\quad \text{Hand}(x, h_x) \land \text{Hand}(y, h_y) \land \\
        &\quad \text{MoveToward}(h_x, h_y) \land \text{Contact}(h_x, h_y) \land \\
        &\quad \text{Shake}(h_x, h_y) \land \\
        &\quad \text{Release}(h_x, h_y) )
    \end{align*}
  \end{itemize}
\item
  \textbf{\textcolor{darkgreen}{Probabilistic nature of knowledge}}

  \begin{itemize}
  \tightlist
  \item
    Example in medicine: \emph{``Fever, cough, and fatigue could
    indicate flu, COVID-19, or another illness.''}
  \end{itemize}
\item
  \textbf{\textcolor{blue}{Scalability challenges}}

  \begin{itemize}
  \tightlist
  \item
    Large problems may need heuristics for practical solutions
  \end{itemize}
\item

\begin{frame}{Problems of a Rational Agent}
\phantomsection\label{problems-of-a-rational-agent}

  \begin{itemize}
  \tightlist
  \item
    A rational agent aims for:

    \begin{itemize}
    \tightlist
    \item
      The best outcome in a deterministic setup
    \item
      The best expected outcome under uncertainty
    \end{itemize}
  \end{itemize}
\item
  \textbf{\textcolor{orange}{Best}} is determined by the objective
  function:

  \begin{itemize}
  \tightlist
  \item

  \end{itemize}
\item
  Omniscience vs \textbf{\textcolor{darkgreen}{no-regrets}}

  \begin{itemize}
  \tightlist
  \item
    Best based on available information
  \end{itemize}
\item
  Sometimes \textbf{\textcolor{teal}{no provably correct action}} exists

  \begin{itemize}
  \tightlist
  \item
    Yet, an action must be taken
  \end{itemize}
\item
  Even \textbf{\textcolor{cyan}{with perfect information}} rationality
  can't be feasible due to:

  \begin{itemize}
  \tightlist
  \item
    Cost of acquiring all data (e.g., in medicine)
  \item
    Computational demands
  \end{itemize}
\item
  Perfect good enough vs perfect

  \begin{itemize}
  \tightlist
  \item
    Acting appropriately (``satisficing'')
  \end{itemize}

  \begin{itemize}

    Computer vision
  \item
    Speech recognition
  \item
    Natural language processing
  \end{itemize}

\begin{frame}{Limits of ML Compared to Human Intelligence}
\phantomsection\label{limits-of-ml-compared-to-human-intelligence}
\begin{itemize}
\tightlist
\item
  \textbf{\textcolor{red}{AI differs from human intelligence}}

  \begin{itemize}
  \tightlist
  \item
    Machines don't learn like humans (e.g., LLMs)
  \end{itemize}
\item
  \textbf{\textcolor{orange}{Fragility to input variations}}

  \begin{itemize}
  \tightlist
  \item
    ML models fail with slight input distortions

    \begin{itemize}
    \tightlist
    \item
      Adversarial attacks cause misclassification by altering one pixel
    \item
      A model trained for a video game may fail if the screen is
      slightly rotated; humans continue effortlessly
    \end{itemize}
  \end{itemize}
\item
  \textbf{\textcolor{darkgreen}{Lack of transfer learning}}

  \begin{itemize}
  \tightlist
  \item
    ML systems cannot apply knowledge across domains without retraining
  \end{itemize}
\item
  \textbf{\textcolor{teal}{Massive data and compute requirements}}

  \begin{itemize}
  \tightlist
  \item
    ML requires enormous datasets and computational resources

    \begin{itemize}
    \tightlist
    \item
      A teenager learns to drive in hours
    \item
      Self-driving systems need billions of compute hours and extensive
      data
    \end{itemize}
  \end{itemize}
\item
  \textbf{\textcolor{cyan}{Poor common sense and reasoning}}

  \begin{itemize}
  \tightlist
  \item
    ML lacks built-in world knowledge and intuitive logic
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Limits of ML Compared to Human Intelligence}

\tightlist
\item
  \textbf{\textcolor{red}{Opaque decision-making}}

  \item

  \end{itemize}

  \textbf{\textcolor{darkgreen}{Dependence on narrow objectives}}

  \tightlist
  \item

    \begin{itemize}
    \tightlist
    \item

    \end{itemize}
  \end{itemize}

  \textbf{\textcolor{blue}{Susceptibility to bias and data quality}}

  \begin{itemize}

  \end{itemize}
\item
  \textbf{\textcolor{violet}{Lack of embodiment and physical interaction}}

  \item

  \end{itemize}

\begin{frame}{The 3 Machine Learning Assumptions}
\phantomsection\label{the-3-machine-learning-assumptions}
\begin{itemize}
\tightlist
\item
  In practice, ML involves solving a practical problem by:

  \begin{itemize}
  \tightlist
  \item
    Gathering a dataset
  \item
    Building a statistical model from the dataset algorithmically
  \end{itemize}
\item
  The \textbf{\textcolor{red}{three assumptions}} of machine learning

  \begin{itemize}
  \tightlist
  \item
    A \textbf{\textcolor{orange}{pattern exists}}
  \item
    Pattern cannot be
    \textbf{\textcolor{darkgreen}{precisely defined mathematically}}
  \item
    \textbf{\textcolor{teal}{Data is available}}
  \end{itemize}
\item
  Which ML assumption is \textbf{\textcolor{cyan}{essential}}?

  \begin{itemize}
  \tightlist
  \item
    A pattern exists

    \begin{itemize}
    \tightlist
    \item
      If no pattern, try learning, measure effectiveness, conclude it
      doesn't work
    \end{itemize}
  \item
    Pattern cannot be precisely defined mathematically

    \begin{itemize}
    \tightlist
    \item
      If solution is direct, ML not recommended, but may still apply
    \end{itemize}
  \item
    Data is available

    \begin{itemize}
    \tightlist
    \item
      Without data, no progress can be made
    \item
      \textbf{\textcolor{blue}{Data is crucial}}
    \end{itemize}
  \end{itemize}
\end{itemize}
\end{frame}
