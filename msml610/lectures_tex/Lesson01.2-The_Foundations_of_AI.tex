\begin{frame}{AI and Philosophy (1/2)}
\phantomsection\label{ai-and-philosophy-12}
\begin{itemize}
\tightlist
\item
  \textbf{\textcolor{red}{\textcolor{black}{\text{Can formal rules be used to draw valid conclusions?}}}}

  \begin{itemize}
  \tightlist
  \item
    \textbf{\textcolor{orange}{Reasoning}}

    \begin{itemize}
    \tightlist
    \item
      Logic studies rules of proper reasoning
    \item
      Aristotle (400 BCE) formulated laws governing the rational mind
    \item
      Machines were built for arithmetic operations (e.g., Pascaline,
      1600)
    \end{itemize}
  \item
    \textbf{\textcolor{darkgreen}{Rationalism}}

    \begin{itemize}
    \tightlist
    \item
      Use reasoning to understand the world
    \end{itemize}
  \end{itemize}
\item
  \textbf{\textcolor{teal}{\textcolor{black}{\text{How does the mind arise from a physical brain?}}}}

  \begin{itemize}
  \tightlist
  \item
    \textbf{\textcolor{cyan}{Dualism}}

    \begin{itemize}
    \tightlist
    \item
      Nature follows physical laws
    \item
      Part of the human mind (``the soul'') is exempt from physical laws
    \end{itemize}
  \item
    \textbf{\textcolor{blue}{Materialism}}

    \begin{itemize}
    \tightlist
    \item
      The mind is a physical system, following the laws of physics
    \item
      Where is free will? Free will is the perception of available
      choices
    \end{itemize}
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{AI and Philosophy (2/2)}
\phantomsection\label{ai-and-philosophy-22}
\begin{columns}[T]
\begin{column}{0.75\textwidth}
\begin{itemize}
\tightlist
\item
  \textbf{\textcolor{red}{\textcolor{black}{\text{What does knowledge come from?}}}}

  \begin{itemize}
  \tightlist
  \item
    \textbf{\textcolor{orange}{Empiricism}}

    \begin{itemize}
    \tightlist
    \item
      Knowledge acquired via senses
    \item
      E.g., learn that trees are green by looking at them
    \end{itemize}
  \item
    \textbf{\textcolor{darkgreen}{Induction}}

    \begin{itemize}
    \tightlist
    \item
      General rules from associations
    \item
      E.g., many swans are white, infer all swans are white
    \end{itemize}
  \item
    \textbf{\textcolor{teal}{Logical Positivism}}

    \begin{itemize}
    \tightlist
    \item
      Knowledge as logical theories linked to observations
    \item
      E.g., scientific hypotheses connected to experimental data
    \end{itemize}
  \end{itemize}
\end{itemize}
\end{column}

\begin{column}{0.2\textwidth}
\includegraphics{msml610/lectures_source/figures/L01.2.black_swans.jpg}
\end{column}
\end{columns}

\begin{itemize}
\tightlist
\item
  \textbf{\textcolor{cyan}{\textcolor{black}{\text{How does knowledge lead to action?}}}}

  \begin{itemize}
  \tightlist
  \item
    \textbf{\textcolor{blue}{Utilitarianism}}

    \begin{itemize}
    \tightlist
    \item
      Actions justified by logic connecting goals and outcomes
    \end{itemize}
  \item
    \textbf{\textcolor{violet}{Consequentialism}}

    \begin{itemize}
    \tightlist
    \item
      Right or wrong determined by action's expected outcomes
    \item
      E.g., ``If you kill, you will go to jail''
    \end{itemize}
  \item
    \textbf{\textcolor{magenta}{Deontological ethics}}

    \begin{itemize}
    \tightlist
    \item
      ``Right actions'' based on universal laws, not outcomes
    \item
      E.g., ``don't kill'', ``don't lie''
    \end{itemize}
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{AI and Cognitive Psychology}
\phantomsection\label{ai-and-cognitive-psychology}
\begin{itemize}
\tightlist
\item
  \textbf{\textcolor{red}{\textcolor{black}{\text{How do humans think and act?}}}}

  \begin{itemize}
  \tightlist
  \item
    \textbf{\textcolor{darkgreen}{Cognitive psychology}}

    \begin{itemize}
    \tightlist
    \item
      Brain is an information-processing device
    \item
      Stimuli translated into internal representation
    \item
      Representation manipulated by cognitive processes to derive new
      internal representations (``beliefs'')
    \item
      Representations turned into actions (``goals'')
    \end{itemize}
  \item
    \textbf{\textcolor{blue}{Cognitive science}}

    \begin{itemize}
    \tightlist
    \item
      Use computer models to address memory, language, and logic

    \item
      Dual / opposite of AI
    \end{itemize}
  \item
    \textbf{\textcolor{violet}{Human-computer interaction}}

    \begin{itemize}
    \tightlist
    \item
      Computers augment human abilities
    \item
      From artificial intelligence (AI) to intelligence augmentation
      (IA)
    \end{itemize}
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{AI and Mathematics}
\phantomsection\label{ai-and-mathematics}
\begin{itemize}
\tightlist
\item
  \textbf{\textcolor{red}{\textcolor{black}{\text{What are the formal rules to draw valid conclusions?}}}}

  \begin{itemize}
  \tightlist
  \item
    \textbf{\textcolor{orange}{Formal logic}}

    \begin{itemize}
    \tightlist
    \item
      Logical deduction rules (Boole, 1850)
    \item
      First-order logic includes objects and relations (Frege, 1879)
    \end{itemize}
  \item
    \textbf{\textcolor{darkgreen}{Limits to deduction}}

    \begin{itemize}
    \tightlist
    \item
      Some statements are ``undecidable''
    \item
      Incompleteness theorem: in any formal theory true statements exist
      that cannot be proved (Godel, 1931)
    \end{itemize}
  \end{itemize}
\item
  \textbf{\textcolor{teal}{\textcolor{black}{\text{How do we reason with uncertain information?}}}}

  \begin{itemize}
