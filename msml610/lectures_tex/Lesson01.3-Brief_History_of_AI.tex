\begin{frame}{The Beginning (1943-1956)}
\phantomsection\label{the-beginning-1943-1956}
\begin{itemize}
\tightlist
\item
  \textbf{\textcolor{red}{Artificial neuron}}

  \begin{itemize}
  \tightlist
  \item
    Model (McCullock-Pitts 1943) based on:

    \begin{itemize}
    \tightlist
    \item
      Brain physiology
    \item
      Propositional logic
    \end{itemize}
  \item
    Compute any function with connected neurons

    \begin{itemize}
    \tightlist
    \item
      Neuron on/off based on stimulation from neighboring neurons
    \item
      Implement logical AND, OR, NOT with simple neuron networks
    \end{itemize}
  \end{itemize}
\item
  \textbf{\textcolor{darkgreen}{Alan Turing, 1947}}

  \begin{itemize}
  \tightlist
  \item
    Turing test, machine learning, reinforcement learning
  \item
    Create human-level AI

    \begin{itemize}
    \tightlist
    \item
      Develop learning algorithms
    \item
      Teach machine like a child
    \end{itemize}
  \end{itemize}
\item
  \textbf{\textcolor{blue}{Birth of AI}}

  \begin{itemize}
  \tightlist
  \item
    McCarthy organized first AI workshop (1956)
  \item
    The Logic Theorist (Newell and Simon, 1956)

    \begin{itemize}
    \tightlist
    \item
      Programs to ``think non-numerically'' and prove theorems
    \end{itemize}
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Enthusiasm and Great Expectations (1952-1969)}
\phantomsection\label{enthusiasm-and-great-expectations-1952-1969}
\begin{itemize}
\tightlist
\item
  \textbf{\textcolor{red}{Early years of AI were full of successes}}

  \begin{itemize}
  \tightlist
  \item
    Before computers could only do arithmetics
  \item
    \emph{``A machine can never do X (e.g., games, puzzles, IQ tests,
    \ldots)''}

    \begin{itemize}
    \tightlist
    \item
      AI researchers showed machines could do one \(X\) after another
    \end{itemize}
  \end{itemize}
\item
  \textbf{\textcolor{orange}{General Problem Solver}}

  \begin{itemize}
  \tightlist
  \item
    Imitate human problem-solving
  \item
    Consider sub-goals and possible actions
  \end{itemize}
\item
  \textbf{\textcolor{darkgreen}{Program learned to play checkers}}

  \begin{itemize}
  \tightlist
  \item
    Use reinforcement learning from victories and mistakes
  \end{itemize}
\item
  \textbf{\textcolor{teal}{Lisp (1958)}}

  \begin{itemize}
  \tightlist
  \item
    High-level language used for 30 years in AI
  \end{itemize}

  \textbf{\textcolor{cyan}{First neural network}}

  \begin{itemize}
  \tightlist
  \item
    3000 vacuum tubes for 40 neurons
  \item
    Minsky (1959)
  \end{itemize}
\item
  \textbf{\textcolor{blue}{MIT and Stanford}}

  \begin{itemize}
  \tightlist
  \item
    Minsky at MIT

    \begin{itemize}
    \tightlist
    \item
      Focus on neural network
    \end{itemize}
  \item
    McCarthy at Stanford

    \begin{itemize}
    \tightlist
    \item
      Focus on representation, logic
    \end{itemize}
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{First AI winter (1975-1980)}
\phantomsection\label{first-ai-winter-1975-1980}
\begin{itemize}
\item
  \textbf{\textcolor{red}{Early successes}} of AI set high expectations
\item
  In 1965-1975 AI didn't succeed on
  \textbf{\textcolor{darkgreen}{real problems}} due to:

  \begin{itemize}
  \tightlist
  \item
    Solutions were based on human problem-solving methods
  \item
    Difficulty handling ``combinatorial explosion''

    \begin{itemize}
    \tightlist
    \item
      Theorem proving handles small problems with brute force, but
      doesn't scale
    \end{itemize}
  \item
    Neural networks needed:

    \begin{itemize}
    \tightlist
    \item
      Algorithms (e.g., backpropagation)
    \item
      Compute power
    \item
      Data
    \end{itemize}
  \end{itemize}
\item
  \textbf{\textcolor{blue}{First AI winter}}

  \begin{itemize}
  \tightlist
  \item
    Research funding and enthusiasm dropped significantly
  \item
    Slow AI progress through late 1970s
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Expert Systems (1980-1990)}
\phantomsection\label{expert-systems-1980-1990}
\begin{itemize}
\tightlist
\item
  \textbf{\textcolor{red}{Expert systems}}

  \begin{itemize}
  \tightlist
  \item
    Aka ``knowledge-based systems''
  \item
    Combine weak methods with extensive domain knowledge as rules
  \item
    Use inference engines to apply rules to facts
  \item
    E.g., rule-based systems, logic programming (e.g., Prolog)
  \end{itemize}
\item
  \textbf{\textcolor{darkgreen}{Weak AI}}

  \begin{itemize}
  \tightlist
  \item
    Aka narrow AI
  \item
    Performs specific tasks, not general reasoning
  \item
    Operates in a limited, well-defined domain
  \item
    Uses ``weak methods'' (search, logic) that struggle to scale
  \end{itemize}
\item
  \textbf{\textcolor{blue}{Commercial adoption and industry growth}}

  \begin{itemize}
  \tightlist
  \item
    AI shifted to practical applications
  \item
    Major US corporations deployed expert systems
  \item
    AI emerged as a commercial industry
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Second AI Winter (late 1980-early 1990)}
\phantomsection\label{second-ai-winter-late-1980-early-1990}
\begin{itemize}
\item
  \textbf{\textcolor{red}{Hype in expert systems}} didn't deliver
\item
  \textbf{\textcolor{darkgreen}{Reasons}}

  \begin{itemize}
  \tightlist

\begin{frame}{Cooperative Inverse Reinforcement Learning}
\phantomsection\label{cooperative-inverse-reinforcement-learning}

\item

  couch, observes the messy table, and starts watching TV
\item
  \textbf{\textcolor{darkgreen}{Inference}}: AI infers:

  \begin{itemize}
  \tightlist
  \item
    GP is tired and wants to relax
  \item
    Messy coffee table bothers him
  \end{itemize}
\item
  \textbf{\textcolor{blue}{Action}}: AI:

  \begin{itemize}
  \tightlist
  \item
    Fetches a glass of water
  \item
    Tidies up the coffee table without disturbing GP
  \end{itemize}
\item
  \textbf{\textcolor{violet}{Feedback loop}}: AI monitors GP's reactions

  \begin{itemize}
  \tightlist
  \item
    If GP is relaxed and happy, AI understanding is reinforced
  \item
    If GP is not happy, AI adjusts actions and improves inference
  \end{itemize}
\end{itemize}
